sse =[]                            # this is a way of finding Sum of Square Error
k_range = range(1,11)
for i in k_range:
    km = KMeans(n_clusters=i)
    km.fit(df)
    sse.append(km.inertia_)        # 'inertia' give us the SSE for each 'K'
